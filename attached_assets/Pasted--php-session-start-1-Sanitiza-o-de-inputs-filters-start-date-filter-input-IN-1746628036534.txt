<?php
session_start();

// 1. Sanitização de inputs
$filters = [
    'start_date' => filter_input(INPUT_GET, 'start_date', FILTER_SANITIZE_STRING) ?: '',
    'end_date'   => filter_input(INPUT_GET, 'end_date', FILTER_SANITIZE_STRING)   ?: '',
    'model_id'   => filter_input(INPUT_GET, 'model_id', FILTER_VALIDATE_INT)      ?: '',
    'status'     => filter_input(INPUT_GET, 'status', FILTER_SANITIZE_STRING)     ?: '',
    'company_id' => ($_SESSION['role'] === 'admin')
        ? (filter_input(INPUT_GET, 'company_id', FILTER_VALIDATE_INT) ?: '')
        : $_SESSION['company_id'],
];

// 2. Busca de empresas com prepared statement
$companies = [];
if ($_SESSION['role'] === 'admin') {
    $stmt = $pdo->prepare("SELECT id, name FROM companies ORDER BY name");
    $stmt->execute();
    $companies = $stmt->fetchAll(PDO::FETCH_ASSOC);
}

// 3. Paginação
$currentPage = filter_input(INPUT_GET, 'page_num', FILTER_VALIDATE_INT) ?: 1;
$perPage     = 10;

// 4. Busca de pedidos (certifique-se de usar prepared statements dentro de getOrders)
$orders = getOrders($pdo, $filters, $currentPage, $perPage);
?>

<div class="card mb-4">
    <div class="card-header">
        <h2 class="card-title h5 mb-0">Filtros</h2>
    </div>
    <div class="card-body">
        <form method="get" action="" class="row g-3">
            <input type="hidden" name="page" value="order_listing">
            <!-- Reseta paginação ao aplicar filtros -->
            <input type="hidden" name="page_num" value="1">

            <div class="col-md-3">
                <label for="start_date" class="form-label">Data Inicial</label>
                <input
                    type="date"
                    class="form-control"
                    id="start_date"
                    name="start_date"
                    value="<?php echo htmlspecialchars($filters['start_date'], ENT_QUOTES); ?>"
                >
            </div>
            <div class="col-md-3">
                <label for="end_date" class="form-label">Data Final</label>
                <input
                    type="date"
                    class="form-control"
                    id="end_date"
                    name="end_date"
                    value="<?php echo htmlspecialchars($filters['end_date'], ENT_QUOTES); ?>"
                >
            </div>
            <div class="col-md-4">
                <label for="model_id" class="form-label">Modelo</label>
                <select class="form-select" id="model_id" name="model_id">
                    <option value="">Todos os Modelos</option>
                    <?php foreach (getProductModels($pdo) as $model): ?>
                        <option
                            value="<?php echo (int)$model['id']; ?>"
                            <?php echo ($filters['model_id'] == $model['id']) ? 'selected' : ''; ?>
                        >
                            <?php echo htmlspecialchars($model['name'], ENT_QUOTES); ?>
                        </option>
                    <?php endforeach; ?>
                </select>
            </div>
            <div class="col-md-3">
                <label for="status" class="form-label">Status</label>
                <select class="form-select" id="status" name="status">
                    <option value="">Todos os Status</option>
                    <?php
                    $statuses = ['Em produção','Gravado','Separado','Enviado','Entregue'];
                    foreach ($statuses as $st): ?>
                        <option
                            value="<?php echo $st; ?>"
                            <?php echo ($filters['status'] === $st) ? 'selected' : ''; ?>
                        >
                            <?php echo $st; ?>
                        </option>
                    <?php endforeach; ?>
                </select>
            </div>
            <?php if ($_SESSION['role'] === 'admin'): ?>
            <div class="col-md-3">
                <label for="company_id" class="form-label">Empresa</label>
                <select class="form-select" id="company_id" name="company_id">
                    <option value="">Todas as Empresas</option>
                    <?php foreach ($companies as $company): ?>
                        <option
                            value="<?php echo (int)$company['id']; ?>"
                            <?php echo ($filters['company_id'] == $company['id']) ? 'selected' : ''; ?>
                        >
                            <?php echo htmlspecialchars($company['name'], ENT_QUOTES); ?>
                        </option>
                    <?php endforeach; ?>
                </select>
            </div>
            <?php endif; ?>
            <div class="col-md-2 d-flex align-items-end">
                <div class="d-flex gap-2 w-100">
                    <button type="submit" class="btn btn-primary flex-grow-1">
                        <i class="fas fa-filter me-1"></i> Filtrar
                    </button>
                    <a href="index.php?page=order_listing" class="btn btn-outline-secondary">
                        <i class="fas fa-undo"></i>
                    </a>
                </div>
            </div>
        </form>
    </div>
</div>

<div class="card">
    <div class="card-header">
        <h2 class="card-title h5 mb-0">Listagem de Pedidos</h2>
    </div>
    <div class="card-body">
        <div class="table-responsive">
            <table class="table table-hover">
                <thead>
                    <tr>
                        <th>Criado por</th>
                        <th>Cliente</th>
                        <th>Modelo</th>
                        <th>Tipo de Metal</th>
                        <th>Status</th>
                        <th>Data de Entrega</th>
                        <th>Imagem</th>
                        <th>Ações</th>
                    </tr>
                </thead>
                <tbody>
                    <?php if (empty($orders['data'])): ?>
                    <tr>
                        <!-- colspan ajustado para 8 colunas -->
                        <td colspan="8" class="text-center py-4">
                            <p class="text-muted mb-0">Nenhum pedido encontrado. Crie um novo pedido para começar.</p>
                        </td>
                    </tr>
                    <?php else: ?>
                        <?php foreach ($orders['data'] as $order): ?>
                        <tr>
                            <td><?php echo htmlspecialchars($order['username'] ?? 'N/A', ENT_QUOTES); ?></td>
                            <td><?php echo htmlspecialchars($order['client'] ?? 'N/A', ENT_QUOTES); ?></td>
                            <td><?php echo htmlspecialchars($order['model'], ENT_QUOTES); ?></td>
                            <td><?php echo htmlspecialchars($order['metal_type'], ENT_QUOTES); ?></td>
                            <td>
                                <?php
                                $statusColors = [
                                    'Em produção'=>'bg-primary',
                                    'Gravado'=>'bg-info',
                                    'Separado'=>'bg-warning',
                                    'Enviado'=>'bg-success',
                                    'Entregue'=>'bg-secondary'
                                ];
                                $color = $statusColors[$order['status']] ?? 'bg-secondary';
                                ?>
                                <span class="badge <?php echo $color; ?>">
                                    <?php echo htmlspecialchars($order['status'], ENT_QUOTES); ?>
                                </span>
                            </td>
                            <td><?php echo formatDate($order['delivery_date']); ?></td>
                            <td>
                                <?php
                                $images = json_decode($order['image_urls'] ?? '[]', true);
                                $first = $images[0] ?? '';
                                if ($first):
                                ?>
                                <div class="hover-card">
                                    <button
                                        class="btn btn-sm btn-outline-secondary image-preview-link"
                                        data-image-url="<?php echo htmlspecialchars($first, ENT_QUOTES); ?>"
                                    >
                                        <i class="fas fa-image"></i>
                                    </button>
                                    <div class="hover-card-content">
                                        <img
                                            src="<?php echo htmlspecialchars($first, ENT_QUOTES); ?>"
                                            alt="Order image"
                                            class="img-fluid rounded"
                                            onerror="this.onerror=null;this.src='assets/images/no-image.png';"
                                        >
                                    </div>
                                </div>
                                <?php else: ?>
                                    <span class="text-muted">Sem imagem</span>
                                <?php endif; ?>
                            </td>
                            <td>
                                <!-- botões de ação omitidos para brevidade; mantenha o seu código existente -->
                            </td>
                        </tr>
                        <?php endforeach; ?>
                    <?php endif; ?>
                </tbody>
            </table>

            <?php if (!empty($orders['total'])): ?>
            <div class="d-flex justify-content-center mt-4">
                <nav aria-label="Page navigation">
                    <ul class="pagination">
                        <?php if ($currentPage > 1): ?>
                        <li class="page-item">
                            <a
                                class="page-link"
                                href="?page=order_listing&page_num=<?php echo $currentPage - 1; ?>"
                                aria-label="Previous"
                            >&laquo;</a>
                        </li>
                        <?php endif; ?>

                        <?php for ($i = 1; $i <= $orders['pages']; $i++): ?>
                        <li class="page-item <?php echo $currentPage === $i ? 'active' : ''; ?>">
                            <a class="page-link" href="?page=order_listing&page_num=<?php echo $i; ?>">
                                <?php echo $i; ?>
                            </a>
                        </li>
                        <?php endfor; ?>

                        <?php if ($currentPage < $orders['pages']): ?>
                        <li class="page-item">
                            <a
                                class="page-link"
                                href="?page=order_listing&page_num=<?php echo $currentPage + 1; ?>"
                                aria-label="Next"
                            >&raquo;</a>
                        </li>
                        <?php endif; ?>
                    </ul>
                </nav>
            </div>
            <?php endif; ?>
        </div>
    </div>
</div>
